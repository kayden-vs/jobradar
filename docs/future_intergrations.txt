python# ats_discovery.py — paste any company slug and it finds which ATS they use
import requests

def discover_ats(company_slug: str):
    checks = [
        ("Greenhouse US", f"https://boards.greenhouse.io/v1/boards/{company_slug}/jobs"),
        ("Greenhouse EU", f"https://boards.eu.greenhouse.io/v1/boards/{company_slug}/jobs"),
        ("Lever",         f"https://api.lever.co/v0/postings/{company_slug}"),
        ("Ashby",         f"https://{company_slug}.jobs.ashbyhq.com/api/jobs"),
        ("Workable",      f"https://apply.workable.com/api/v3/accounts/{company_slug}/jobs"),
    ]
    for name, url in checks:
        try:
            method = "POST" if "workable" in url else "GET"
            r = requests.request(method, url, timeout=8,
                                 json={"query":""} if method=="POST" else None)
            if r.status_code == 200:
                data = r.json()
                count = len(data) if isinstance(data, list) else len(data.get("jobs", data.get("results", [])))
                print(f"✅ {name}: {url}  →  {count} jobs")
                return name, url
        except Exception:
            pass
    print(f"❌ No known ATS found for slug: {company_slug}")
    return None, None

python# sources/internshala.py
from scrapling.fetchers import Fetcher  # plain HTTP, no browser

SEARCH_URLS = [
    "https://internshala.com/internships/backend-development-internship/work-from-home-internships/",
    "https://internshala.com/internships/backend-development-internship/",
    "https://internshala.com/internships/web-development-internship/work-from-home-internships/",
    # Golang-specific — use search
    "https://internshala.com/internships/keywords-golang/",
    "https://internshala.com/internships/keywords-go-developer/",
    # Fresher jobs (not just internships)
    "https://internshala.com/jobs/backend-developer-intern-jobs/",
]

def fetch_internshala() -> list[dict]:
    fetcher = Fetcher()
    all_jobs = []
    seen = set()
    
    for url in SEARCH_URLS:
        try:
            page = fetcher.get(url, timeout=15)
            # Internshala job cards
            cards = page.css(".internship_meta, .individual_internship")
            for card in cards:
                title_el   = card.css(".profile")
                company_el = card.css(".company_name")
                loc_el     = card.css(".locations span")
                stipend_el = card.css(".stipend")
                link_el    = card.css("a.view_detail_button, a[href*='/internship/detail/']")
                
                href = link_el[0].attrib.get("href","") if link_el else ""
                job_url = "https://internshala.com" + href if href.startswith("/") else href
                if job_url in seen:
                    continue
                seen.add(job_url)
                
                all_jobs.append({
                    "title":     title_el[0].text.strip() if title_el else "",
                    "company":   company_el[0].text.strip() if company_el else "",
                    "location":  loc_el[0].text.strip() if loc_el else "India",
                    "description": "",
                    "url":       job_url,
                    "source":    "internshala",
                    "salary":    stipend_el[0].text.strip() if stipend_el else "",
                    "posted_at": "",
                })
        except Exception as e:
            print(f"Internshala failed for {url}: {e}")
    return all_jobs

LinkedIn Jobs (URL-based, no auth)
LinkedIn public job search is accessible without login if you construct the URL correctly. The trick is using &f_E=1 (entry level) and &f_JT=I (internship) filters:
https://www.linkedin.com/jobs/search/?keywords=backend+golang+intern&location=India&f_E=1,2&f_JT=I,F&f_TPR=r86400
Parameters:

f_E=1,2 — Entry level, Associate (filters out senior roles)
f_JT=I,F — Internship, Full-time
f_TPR=r86400 — Posted in last 24 hours

Use StealthyFetcher here — LinkedIn has bot detection. The page is JS-rendered so DynamicFetcher may be needed.
